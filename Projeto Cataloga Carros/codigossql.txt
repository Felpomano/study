create table carros(
	id integer not null primary key,
	modelo varchar (70) not null,
	ano integer not null,
	tipo varchar (20),
	fabricante varchar (20)
);

create table usuario(
	idusu integer not null primary key,
	nome varchar(70),
	login varchar(30),
	senha varchar(64)
);

create table tipo(
	idtipo integer not null primary key,
	dstipo varchar(70)
);

create table fabricante(
	idfab integer not null primary key,
	dsfab varchar
);
